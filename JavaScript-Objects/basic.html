class Character {
  constructor(name, level, age){
  this.name = name;
  this.level = level;
  this.age = age;
}
}
class Race extends Character{
  constructor(name, level, classs, race, age){
    super(name, level, age);
      this.classs = classs;
      this.race = race;
}
}

var player1 = new Race('Gandalf', 100, 'Wizard', 'Maia');
document.getElementById('inherit-execute').innerHTML =
'Your party leader is a level ' + player1.level +
 ' ' + player1.classs + ' named ' + player1.name;

function Student(fname, lname, year, degree, gpa) {
  const obj = {};

  obj.fname = fname;
  obj.lname = lname;
  obj.year = year;
  obj.degree = degree;
  obj.gpa = 0;
  obj.setGPA = function(input){
  this.gpa = input;
}
  obj.Iam = function() {
  return 'My name is ' + this.fname + ' ' + this.lname
     + ', I am a ' + this.year + ' in ' + this.degree +
     ' with a ' + this.gpa + ' gpa.';
}
return obj;
}

var student = new Student('Steven','Anderson', 
      'Senior', 'Software Engineering');
var input1 = document.getElementById('input1').value;
student.setGPA(input1);

document.getElementById('inst-execute').innerHTML =
student.Iam();
